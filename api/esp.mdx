## Toggle

</ResponseField>

## Colors

<ResponseField name="ESP.SetColor(type, r, g, b)" type="void">
  Set colors for specific ESP elements.
  <ParamField path="type" type="string" required>
    `"enemy"`, `"skeleton"`, or `"name"`
  </ParamField>
  <ParamField path="r" type="number" required>Red (0–255)</ParamField>
  <ParamField path="g" type="number" required>Green (0–255)</ParamField>
  <ParamField path="b" type="number" required>Blue (0–255)</ParamField>
</ResponseField>

```lua
ESP.SetColor("enemy", 255, 50, 50)    -- red boxes
ESP.SetColor("skeleton", 255, 255, 0) -- yellow skeleton
ESP.SetColor("name", 255, 255, 255)   -- white names
```

## Custom flags

Add custom text flags that appear next to player names in ESP.

<ResponseField name="ESP.SetFlag(playerName, flagText, r, g, b)" type="void">
  Add a custom flag to a player.
  <ParamField path="playerName" type="string" required>Player's username</ParamField>
  <ParamField path="flagText" type="string" required>Flag text to display</ParamField>
  <ParamField path="r" type="number" required>Red (0–255)</ParamField>
  <ParamField path="g" type="number" required>Green (0–255)</ParamField>
  <ParamField path="b" type="number" required>Blue (0–255)</ParamField>
</ResponseField>

<ResponseField name="ESP.RemoveFlag(playerName, flagText)" type="void">
  Remove a specific flag from a player.
</ResponseField>

<ResponseField name="ESP.ClearFlags(playerName)" type="void">
  Remove all flags from a specific player.
</ResponseField>

<ResponseField name="ESP.ClearAllFlags()" type="void">
  Remove all custom flags from all players.
</ResponseField>

```lua
-- Mark a suspicious player
ESP.SetFlag("PlayerName", "SUS", 255, 165, 0)

-- Mark high-value targets
local players = Player.GetPlayers()
for _, p in ipairs(players) do
    if p.Health > 90 then
        ESP.SetFlag(p.Name, "FULL HP", 0, 255, 0)
    end
end
```
