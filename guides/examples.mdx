## Toggle ESP with Keybind

A simple toggle — press F5 to flip ESP on/off with a notification.

```lua
while true do
    if Input.IsKeyPressed(0x74) then -- F5
        local state = ESP.Toggle()
        notify(state and "ESP ON" or "ESP OFF", "Toggle", 2, state and 1 or 2)
    end
    wait(0.01)
end
```

## Dynamic FOV Based on Distance

Automatically widens your aimbot FOV when enemies are far away, and tightens it up close for precision.

```lua
Aimbot.Enable()

while true do
    local closest = Player.GetClosest(true)
    if closest then
        local fov = math.max(50, 300 - closest.Distance * 2)
        Aimbot.SetFOV(fov)
    end
    wait(0.05)
end
```

## Auto-Save Player Logs

Snapshots all current players to a timestamped file in your workspace.

```lua
makefolder("logs")

local players = Player.GetPlayers()
local log = ""

for _, p in ipairs(players) do
    log = log .. p.Name
        .. " | HP: " .. p.Health .. "/" .. p.MaxHealth
        .. " | Dist: " .. math.floor(p.Distance) .. "m"
        .. " | Alive: " .. tostring(p.IsAlive)
        .. "\n"
end

writefile("logs/players_" .. os.time() .. ".txt", log)
notify("Saved " .. #players .. " players", "Logger", 2, 1)
```

## Load Remote Script

Fetches and runs a script from a URL. Useful for keeping scripts up-to-date without manually pasting code.

```lua
local url = "https://raw.githubusercontent.com/user/repo/main/script.lua"
local code = HttpGet(url)
local fn, err = loadstring(code)
if fn then
    fn()
    notify("Remote script loaded", "Loader", 2, 1)
else
    notify("Error: " .. err, "Loader", 5, 2)
end
```

## OOP Crosshair

A persistent crosshair that stays on screen without a loop — set it and forget it.

```lua
local size = Game.GetScreenSize()
local cx, cy = size.Width / 2, size.Height / 2
local gap = 3
local len = 10

local lines = {}
local offsets = {
    { -len, 0, -gap, 0 },
    { gap, 0, len, 0 },
    { 0, -len, 0, -gap },
    { 0, gap, 0, len },
}

for _, o in ipairs(offsets) do
    local l = Drawing.new("Line")
    l.From = Vector2.new(cx + o[1], cy + o[2])
    l.To = Vector2.new(cx + o[3], cy + o[4])
    l.Color = Color3.fromRGB(0, 255, 0)
    l.Thickness = 2
    l.Visible = true
    table.insert(lines, l)
end

local dot = Drawing.new("Circle")
dot.Position = Vector2.new(cx, cy)
dot.Radius = 2
dot.Color = Color3.fromRGB(0, 255, 0)
dot.Filled = true
dot.Visible = true

notify("Crosshair created", "Drawing", 2, 1)
```

## JSON Config System

Save and load your settings to a JSON file. Press F9 to save at any time.

```lua
local FILE = "my_settings.json"

local defaults = {
    aimbot = { enabled = false, fov = 150, smooth = 5 },
    esp = { enabled = true, boxes = true, names = true, distance = true },
}

-- Load existing config or create defaults
local cfg
if isfile(FILE) then
    cfg = JSONDecode(readfile(FILE))
    notify("Settings loaded", "Config", 2, 1)
else
    cfg = defaults
    writefile(FILE, JSONEncode(defaults))
    notify("Created default config", "Config", 2, 0)
end

-- Apply the settings
if cfg.aimbot.enabled then
    Aimbot.Enable()
    Aimbot.SetFOV(cfg.aimbot.fov)
    Aimbot.SetSmoothing(cfg.aimbot.smooth)
end

if cfg.esp.enabled then
    ESP.Enable()
    ESP.SetBoxes(cfg.esp.boxes)
    ESP.SetNames(cfg.esp.names)
    ESP.SetDistance(cfg.esp.distance)
end

-- Save on F9
while true do
    if Input.IsKeyPressed(0x78) then -- F9
        writefile(FILE, JSONEncode(cfg))
        notify("Settings saved", "Config", 2, 1)
    end
    wait(0.01)
end
```

## Custom ESP Flags

Automatically tag players with colored flags based on their health and distance.

```lua
ESP.Enable()
ESP.SetFlags(true)

while true do
    local players = Player.GetPlayers()
    for _, p in ipairs(players) do
        if p.IsAlive and not p.IsTeammate then
            -- Clear old flags first
            ESP.ClearFlags(p.Name)

            if p.Distance < 50 then
                ESP.SetFlag(p.Name, "CLOSE", 255, 50, 50)
            end
            if p.Health < 30 then
                ESP.SetFlag(p.Name, "LOW HP", 255, 200, 0)
            end
        end
    end
    wait(0.5)
end
```

## World Labels

Draw text labels at 3D world positions — great for marking spawns, objectives, or points of interest.

```lua
while true do
    local spawns = Game.FindByName("SpawnPoint")
    for _, s in ipairs(spawns) do
        local screen = Game.WorldToScreen(s.Position.X, s.Position.Y + 3, s.Position.Z)
        if screen.OnScreen then
            Draw.Text(screen.X - 25, screen.Y, "SPAWN", 255, 255, 0, 255, true)
        end
    end
    wait(0.01)
end
```

## Event-Driven Overlay

Using `cheat.run()` instead of a manual loop — the draw buffer is handled automatically.

```lua
local startTime = Game.GetTime()

cheat.register("onPaint", function()
    local elapsed = Game.GetTime() - startTime
    local size = Game.GetScreenSize()

    -- Animated pulsing circle
    local radius = 20 + math.sin(elapsed * 3) * 10
    Draw.Circle(size.Width / 2, size.Height / 2, radius, 0, 255, 0, 200, 2)

    -- FPS counter using delta time
    local fps = math.floor(1 / Game.GetDeltaTime())
    Draw.Text(10, 10, "FPS: " .. fps, 255, 255, 255, 255, true)
end)

cheat.run()
```

## Custom Script UI

Build a full settings panel that lives in the aephylia menu.

```lua
-- Create a tab and container
ui.newTab("my_esp", "My ESP")
ui.newContainer("my_esp", "settings", "Settings", { autosize = true })

-- Add widgets
ui.newCheckbox("my_esp", "settings", "Enabled")
ui.newSliderInt("my_esp", "settings", "Render Distance", 50, 1000, 500)
ui.newDropdown("my_esp", "settings", "Box Style", {"Corners", "Full", "3D"}, 1)
ui.newColorpicker("my_esp", "settings", "Box Color", {r=255, g=80, b=80, a=255})

-- React to widget values
while true do
    local enabled = ui.getValue("my_esp", "settings", "Enabled")
    if enabled then
        ESP.Enable()
        local dist = ui.getValue("my_esp", "settings", "Render Distance")
        ESP.SetRenderDistance(dist)

        local style = ui.getValue("my_esp", "settings", "Box Style")
        ESP.SetBoxStyle(style - 1) -- dropdown is 1-based, style is 0-based

        local color = ui.getValue("my_esp", "settings", "Box Color")
        ESP.SetColor("enemy", color.r, color.g, color.b, color.a)
    else
        ESP.Disable()
    end
    wait(0.05)
end
```
